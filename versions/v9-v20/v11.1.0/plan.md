```rust
//! Fav AST → Python ソースコード生成
use crate::frontend::ast::{self, Program};

pub struct Emitter {
    indent: usize,
    buf: String,
    /// インライン helper 関数の連番
    cond_counter: usize,
    match_counter: usize,
    /// import 収集（先頭に出力）
    imports: std::collections::BTreeSet<String>,
}

impl Emitter {
    pub fn new() -> Self {
        Self {
            indent: 0,
            buf: String::new(),
            cond_counter: 0,
            match_counter: 0,
            imports: std::collections::BTreeSet::new(),
        }
    }

    fn push(&mut self, s: &str) {
        self.buf.push_str(s);
    }

    fn indent_str(&self) -> String {
        "    ".repeat(self.indent)
    }

    fn line(&mut self, s: &str) {
        let ind = self.indent_str();
        self.buf.push_str(&format!("{}{}\n", ind, s));
    }

    fn blank(&mut self) {
        self.buf.push('\n');
    }
}

/// 公開エントリポイント — Program → Python ソース文字列
pub fn emit_python(prog: &Program, source_path: &str) -> String {
    let mut e = Emitter::new();
    e.emit_program(prog, source_path)
}

/// テスト用: ソース文字列から直接 Python を生成
pub fn emit_python_str(fav_src: &str) -> String {
    let prog = crate::frontend::parser::parse(fav_src).expect("parse failed");
    emit_python(&prog, "<test>")
}
```

### A-3: 型定義変換

```rust
impl Emitter {
    fn emit_type_def(&mut self, td: &ast::TypeDef) {
        match &td.kind {
            ast::TypeDefKind::Record(fields) => {
                self.line("@dataclass");
                self.line(&format!("class {}:", td.name));
                self.indent += 1;
                if fields.is_empty() {
                    self.line("pass");
                } else {
                    for (fname, fty) in fields {
                        let py_ty = self.map_type(fty);
                        self.line(&format!("{}: {}", fname, py_ty));
                    }
                }
                self.indent -= 1;
                self.blank();
            }
            // Sum type / nominal wrapper は v11.1.0 スコープ外 — コメント出力
            _ => {
                self.line(&format!("# TODO: type {} (sum/nominal) — not yet supported", td.name));
                self.blank();
            }
        }
    }

    fn map_type(&self, ty: &ast::Type) -> String {
        match ty {
            ast::Type::Named(n, args) => match n.as_str() {
                "String"  => "str".to_string(),
                "Int"     => "int".to_string(),
                "Float"   => "float".to_string(),
                "Bool"    => "bool".to_string(),
                "Unit"    => "None".to_string(),
                "List"    => format!("List[{}]", args.first()
                    .map(|a| self.map_type(a)).unwrap_or("Any".to_string())),
                "Option"  => format!("Optional[{}]", args.first()
                    .map(|a| self.map_type(a)).unwrap_or("Any".to_string())),
                "Result"  => "Result".to_string(),
                other     => other.to_string(),
            },
            ast::Type::Generic(n) => n.to_string(),
            ast::Type::Unit       => "None".to_string(),
        }
    }
}
```

```rust
    fn emit_fn_def(&mut self, fd: &ast::FnDef) {
        // エフェクトコメント
        if !fd.effects.is_empty() {
            let eff_names: Vec<&str> = fd.effects.iter().map(|e| match e {
                ast::Effect::Io         => "IO",
                ast::Effect::Aws        => "AWS",
                ast::Effect::Postgres   => "Postgres",
                ast::Effect::Snowflake  => "Snowflake",
                ast::Effect::Http       => "Http",
                ast::Effect::Llm        => "Llm",
                ast::Effect::Network    => "Network",
                ast::Effect::Queue      => "Queue",
                ast::Effect::Cache      => "Cache",
            }).collect();
            self.line(&format!("# effects: {}", eff_names.join(", ")));
        }

        // def シグネチャ
        let params: Vec<String> = fd.params.iter()
            .map(|(n, t)| format!("{}: {}", n, self.map_type(t)))
            .collect();
        let ret_ty = self.map_type(&fd.ret_type);
        self.line(&format!("def {}({}) -> {}:", fd.name, params.join(", "), ret_ty));

        // ボディ
        self.indent += 1;
        self.emit_block_as_body(&fd.body);
        self.indent -= 1;
        self.blank();
    }
}
```

### A-5: ブロック・ステートメント変換

```rust
impl Emitter {
    /// ブロックをステートメント列 + return で出力
    fn emit_block_as_body(&mut self, block: &ast::Block) {
        for stmt in &block.stmts {
            self.emit_stmt(stmt);
        }
        let ret_expr = self.emit_expr_inline(&block.expr);
        self.line(&format!("return {}", ret_expr));
    }

    fn emit_stmt(&mut self, stmt: &ast::Stmt) {
        match stmt {
            ast::Stmt::Bind(b) => {
                if b.name == "_" {
                    let expr = self.emit_expr_inline(&b.expr);
                    self.line(&expr);
                } else {
                    let expr = self.emit_expr_inline(&b.expr);
                    self.line(&format!("{} = {}", b.name, expr));
                }
            }
            ast::Stmt::Expr(e) => {
                let expr = self.emit_expr_inline(e);
                self.line(&expr);
            }
            ast::Stmt::Chain(c) => {
                let expr = self.emit_expr_inline(&c.expr);
                self.line(&format!("{} = {}", c.name, expr));
            }
            ast::Stmt::Yield(_) | ast::Stmt::ForIn(_) => {
                self.line("# TODO: yield/for-in — not yet supported");
            }
        }
    }
}
```

### A-6: 式変換（インライン）

```rust
impl Emitter {
    fn emit_expr_inline(&mut self, expr: &ast::Expr) -> String {
        match expr {
            ast::Expr::Lit(lit, _) => self.emit_lit(lit),
            ast::Expr::Ident(name, _) => name.clone(),
            ast::Expr::BinOp(op, l, r, _) => {
                let lhs = self.emit_expr_inline(l);
                let rhs = self.emit_expr_inline(r);
                let py_op = match op {
                    ast::BinOp::Add => "+",  ast::BinOp::Sub => "-",
                    ast::BinOp::Mul => "*",  ast::BinOp::Div => "/",
                    ast::BinOp::Mod => "%",
                    ast::BinOp::Eq  => "==", ast::BinOp::Neq => "!=",
                    ast::BinOp::Lt  => "<",  ast::BinOp::Lte => "<=",
                    ast::BinOp::Gt  => ">",  ast::BinOp::Gte => ">=",
                    ast::BinOp::And => "and",ast::BinOp::Or  => "or",
                };
                format!("({} {} {})", lhs, py_op, rhs)
            }
            ast::Expr::Apply(func, args, _) => self.emit_apply(func, args),
            ast::Expr::FieldAccess(obj, field, _) => {
                let o = self.emit_expr_inline(obj);
                format!("{}.{}", o, field)
            }
            ast::Expr::If(cond, then_b, else_b, _) => {
                self.emit_if_expr(cond, then_b, else_b.as_deref())
            }
            ast::Expr::Match(scrutinee, arms, _) => {
                self.emit_match_expr(scrutinee, arms)
            }
            ast::Expr::Block(b) => {
                // 単一式ブロックはインライン展開、複数ステートメントは helper 関数に切り出し
                if b.stmts.is_empty() {
                    self.emit_expr_inline(&b.expr)
                } else {
                    self.emit_block_as_helper(b)
                }
            }
            ast::Expr::Closure(params, body, _) => {
                if params.len() == 1 && matches!(body.as_ref(), ast::Expr::Ident(_, _) | ast::Expr::BinOp(_, _, _, _)) {
                    let p = &params[0].0;
                    let b = self.emit_expr_inline(body);
                    format!("lambda {}: {}", p, b)
                } else {
                    self.emit_closure_as_helper(params, body)
                }
            }
            ast::Expr::RecordConstruct(ty_name, fields, _) => {
                let args: Vec<String> = fields.iter()
                    .map(|(_, v)| self.emit_expr_inline(v))
                    .collect();
                format!("{}({})", ty_name, args.join(", "))
            }
            _ => "None  # TODO: unsupported expr".to_string(),
        }
    }
}
```

```rust
impl Emitter {
    fn emit_apply(&mut self, func: &ast::Expr, args: &[ast::Expr]) -> String {
        // FieldAccess(Ident("List"), "concat") 等のパターンを変換
        if let ast::Expr::FieldAccess(obj, method, _) = func {
            if let ast::Expr::Ident(ns, _) = obj.as_ref() {
                let a: Vec<String> = args.iter().map(|a| self.emit_expr_inline(a)).collect();
                match (ns.as_str(), method.as_str()) {
                    // List
                    ("List", "empty")   => return "[]".to_string(),
                    ("List", "concat")  => return format!("({} + {})", a[0], a[1]),
                    ("List", "length")  => return format!("len({})", a[0]),
                    ("List", "filter")  => return format!("[_x for _x in {} if {}(_x)]", a[0], a[1]),
                    ("List", "map")     => return format!("[{}(_x) for _x in {}]", a[1], a[0]),
                    ("List", "first")   => return format!("({0}[0] if {0} else None)", a[0]),
                    ("List", "drop")    => return format!("{}[{}:]", a[0], a[1]),
                    ("List", "nth")     => return format!("({}[{}] if len({0}) > {} else None)", a[0], a[1], a[1]),
                    ("List", "reverse") => return format!("list(reversed({}))", a[0]),
                    // String
                    ("String", "concat")   => return format!("({} + {})", a[0], a[1]),
                    ("String", "length")   => return format!("len({})", a[0]),
                    ("String", "trim")     => return format!("{}.strip()", a[0]),
                    ("String", "contains") => return format!("({} in {})", a[1], a[0]),
                    ("String", "split")    => return format!("{}.split({})", a[0], a[1]),
                    // Int / Float / Bool
                    ("Int",   "to_string") => return format!("str({})", a[0]),
                    ("Float", "to_string") => return format!("str({})", a[0]),
                    ("Bool",  "to_string") => return format!("str({}).lower()", a[0]),
                    // IO — プレースホルダー（v11.3.0 で実変換）
                    ("IO", name) => return format!("_io_{}({})", name, a.join(", ")),
                    // AWS — プレースホルダー（v11.4.0 で実変換）
                    ("AWS", name) => return format!("_aws_{}({})", name, a.join(", ")),
                    // Postgres — プレースホルダー（v11.6.0 で実変換）
                    ("Postgres", name) => return format!("_postgres_{}({})", name, a.join(", ")),
                    _ => {}
                }
            }
        }
        // 通常の関数呼び出し
        let f = self.emit_expr_inline(func);
        let a: Vec<String> = args.iter().map(|a| self.emit_expr_inline(a)).collect();
        format!("{}({})", f, a.join(", "))
    }
}
```

### A-8: match 変換

```rust
impl Emitter {
    fn emit_match_expr(&mut self, scrutinee: &ast::Expr, arms: &[ast::MatchArm]) -> String {
        let sc = self.emit_expr_inline(scrutinee);
        let n = self.match_counter;
        self.match_counter += 1;
        let var = format!("_m{}", n);

        // Option パターン（Some(v) / None）
        let is_option = arms.iter().any(|a| matches!(&a.pat, ast::Pat::Constructor(cn, _) if cn == "Some" || cn == "None"));
        if is_option {
            // インライン三項式に変換できるか確認（各アームが単純式）
            let some_arm = arms.iter().find(|a| matches!(&a.pat, ast::Pat::Constructor(cn, _) if cn == "Some"));
            let none_arm = arms.iter().find(|a| matches!(&a.pat, ast::Pat::Constructor(cn, _) if cn == "None"));
            if let (Some(sa), Some(na)) = (some_arm, none_arm) {
                if let ast::Pat::Constructor(_, inner_pats) = &sa.pat {
                    let bind_name = inner_pats.first()
                        .and_then(|p| if let ast::Pat::Var(v) = p { Some(v.as_str()) } else { None })
                        .unwrap_or("_v");
                    let then_expr = self.emit_expr_inline(&sa.body);
                    let else_expr = self.emit_expr_inline(&na.body);
                    return format!("(lambda {v}: (lambda {b}: {t})({v}) if {v} is not None else {e})({s})",
                        v = var, b = bind_name, t = then_expr, e = else_expr, s = sc);
                }
            }
        }

        // Result パターン（Ok(v) / Err(e)）
        // helper ブロックとして出力（変数を使うため）
        // 呼び出し側の emit_block 内にステートメントとして埋め込む方式を採用
        // インライン不可の場合は "_match_<n>_result" プレースホルダーを返し、
        // Emitter に pending_pre_stmts を積む設計とする（v11.1.0 簡易版: helper 関数化）
        let helper_name = format!("_match_{}", n);
        let mut helper_lines: Vec<String> = Vec::new();
        helper_lines.push(format!("def {}():", helper_name));
        helper_lines.push(format!("    {} = {}", var, sc));
        for (i, arm) in arms.iter().enumerate() {
            let kw = if i == 0 { "if" } else { "elif" };
            let (cond, bindings) = self.arm_condition_and_bindings(arm, &var);
            helper_lines.push(format!("    {} {}:", kw, cond));
            for b in bindings {
                helper_lines.push(format!("        {}", b));
            }
            let body = self.emit_expr_inline(&arm.body);
            helper_lines.push(format!("        return {}", body));
        }
        helper_lines.push(format!("    return None"));
        // helper 関数を buf に追記
        for l in helper_lines {
            self.buf.push_str(&format!("{}\n", l));
        }
        self.blank();
        format!("{}()", helper_name)
    }

    fn arm_condition_and_bindings(&self, arm: &ast::MatchArm, var: &str) -> (String, Vec<String>) {
        match &arm.pat {
            ast::Pat::Constructor(cn, inner) => match cn.as_str() {
                "Ok" => {
                    let bind = inner.first()
                        .and_then(|p| if let ast::Pat::Var(v) = p { Some(format!("    {} = {}.value", v, var)) } else { None })
                        .unwrap_or_default();
                    (format!("isinstance({}, Ok)", var), bind.lines().map(|s| s.trim().to_string()).filter(|s| !s.is_empty()).collect())
                }
                "Err" => {
                    let bind = inner.first()
                        .and_then(|p| if let ast::Pat::Var(v) = p { Some(format!("{} = {}.error", v, var)) } else { None })
                        .unwrap_or_default();
                    (format!("isinstance({}, Err)", var), if bind.is_empty() { vec![] } else { vec![bind] })
                }
                "Some" => {
                    let bind = inner.first()
                        .and_then(|p| if let ast::Pat::Var(v) = p { Some(format!("{} = {}", v, var)) } else { None })
                        .unwrap_or_default();
                    (format!("{} is not None", var), if bind.is_empty() { vec![] } else { vec![bind] })
                }
                "None" => (format!("{} is None", var), vec![]),
                other  => (format!("isinstance({}, {})", var, other), vec![]),
            },
            ast::Pat::Var(v) => (format!("True  # wildcard {}", v), vec![format!("{} = {}", v, var)]),
            ast::Pat::Wild   => ("True  # _".to_string(), vec![]),
            ast::Pat::Lit(l) => (format!("{} == {}", var, self.emit_lit_val(l)), vec![]),
        }
    }
}
```

### B-1: `driver.rs` に `cmd_transpile` 追加

```rust
pub fn cmd_transpile(args: &[String]) -> Result<(), String> {
    let mut target: Option<String> = None;
    let mut out_path: Option<String> = None;
    let mut input_path: Option<String> = None;
    let mut i = 0;
    while i < args.len() {
        match args[i].as_str() {
            "--target" => { i += 1; target = args.get(i).cloned(); }
            "--out"    => { i += 1; out_path = args.get(i).cloned(); }
            _          => { input_path = Some(args[i].clone()); }
        }
        i += 1;
    }
    let target = target.ok_or("error: --target is required")?;
    if target != "python" {
        return Err(format!("error: unsupported target: {} (supported: python)", target));
    }
    let input = input_path.ok_or("error: input file required")?;
    if !std::path::Path::new(&input).exists() {
        return Err(format!("error: file not found: {}", input));
    }
    let src = std::fs::read_to_string(&input)
        .map_err(|e| format!("error: {}", e))?;
    let prog = crate::frontend::parser::parse(&src)
        .map_err(|e| format!("parse error: {}", e))?;
    let py_src = crate::emit_python::emit_python(&prog, &input);
    let out = out_path.unwrap_or_else(|| input.replace(".fav", ".py"));
    std::fs::write(&out, py_src)
        .map_err(|e| format!("error writing {}: {}", out, e))?;
    println!("transpiled: {} -> {}", input, out);
    Ok(())
}
```

```rust
mod v11100_tests {
    use super::*;

    #[test]
    fn transpile_dataclass_simple() {
        let src = "type Point = {\n  x: Int\n  y: Int\n}";
        let out = emit_python_str(src);
        assert!(out.contains("@dataclass"), "missing @dataclass:\n{}", out);

        assert!(out.contains("x: int"), "missing x: int:\n{}", out);
        assert!(out.contains("y: int"), "missing y: int:\n{}", out);

    fn transpile_fn_basic() {
        let src = "fn add(a: Int, b: Int) -> Int { a + b }";
        let out = emit_python_str(src);
        assert!(out.contains("def add(a: int, b: int) -> int:"), "missing def:\n{}", out);

    fn transpile_bind_desugars() {
        let src = "fn f(x: Int) -> Int { bind a <- double(x)  a }";

  }
```
